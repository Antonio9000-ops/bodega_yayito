# SCRIPT DE AUTOMATIZACIÓN LÓGICA: PROCEDIMIENTO ALMACENADO Y TRIGGER DEL KARDEX
script_trigger_kardex = """
-- 1. Crear la función que procesa el movimiento logístico
CREATE OR REPLACE FUNCTION procesar_kardex_y_stock()
RETURNS TRIGGER AS $$
DECLARE
    v_ultimo_saldo_cant INT := 0;
    v_ultimo_saldo_valor DECIMAL(12,2) := 0.00;
    v_nuevo_saldo_cant INT := 0;
    v_nuevo_saldo_valor DECIMAL(12,2) := 0.00;
    v_cant_entrada INT := 0;
    v_costo_u_entrada DECIMAL(12,2) := 0.00;
    v_cant_salida INT := 0;
    v_costo_u_salida DECIMAL(12,2) := 0.00;
BEGIN
    -- Obtener los saldos anteriores del producto desde el último registro del Kardex
    SELECT saldo_cant, saldo_valor
    INTO v_ultimo_saldo_cant, v_ultimo_saldo_valor
    FROM kardex
    WHERE id_producto = NEW.id_producto
    ORDER BY fecha DESC, id DESC
    LIMIT 1;

    -- Si no existe un registro previo en el Kardex, inicializamos en 0
    IF NOT FOUND THEN
        v_ultimo_saldo_cant := 0;
        v_ultimo_saldo_valor := 0.00;
    END IF;

    -- Separar la lógica según el tipo de movimiento logístico
    IF NEW.tipo_movimiento = 'ENTRADA' THEN
        v_cant_entrada := NEW.cantidad;
        v_costo_u_entrada := NEW.costo_operacion;

        -- Cálculos del nuevo saldo (Suma inventario y suma valorizado)
        v_nuevo_saldo_cant := v_ultimo_saldo_cant + v_cant_entrada;
        v_nuevo_saldo_valor := v_ultimo_saldo_valor + (v_cant_entrada * v_costo_u_entrada);

    ELSIF NEW.tipo_movimiento = 'SALIDA' THEN
        v_cant_salida := NEW.cantidad;

        -- En salidas, se usa el Costo Promedio Ponderado para valuar la salida
        IF v_ultimo_saldo_cant > 0 THEN
            v_costo_u_salida := v_ultimo_saldo_valor / v_ultimo_saldo_cant;
        ELSE
            v_costo_u_salida := NEW.costo_operacion; -- Salvaguarda por si no hay stock previo
        END IF;

        -- Cálculos del nuevo saldo (Resta inventario y resta valorizado)
        v_nuevo_saldo_cant := v_ultimo_saldo_cant - v_cant_salida;
        v_nuevo_saldo_valor := v_ultimo_saldo_valor - (v_cant_salida * v_costo_u_salida);
    END IF;

    -- 2. Insertar de forma automática la fila en la tabla Kardex
    INSERT INTO kardex (
        id_producto, id_movimiento, fecha,
        cant_entrada, costo_u_entrada,
        cant_salida, costo_u_salida,
        saldo_cant, saldo_valor
    ) VALUES (
        NEW.id_producto, NEW.id, NEW.fecha,
        v_cant_entrada, v_costo_u_entrada,
        v_cant_salida, v_costo_u_salida,
        v_nuevo_saldo_cant, v_nuevo_saldo_valor
    );

    -- 3. Actualizar el stock actual del producto en la tabla productos de forma síncrona
    UPDATE productos
    SET stock_actual = v_nuevo_saldo_cant,
        costo_promedio = CASE WHEN v_nuevo_saldo_cant > 0 THEN (v_nuevo_saldo_valor / v_nuevo_saldo_cant) ELSE costo_promedio END
    WHERE id = NEW.id_producto;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- 2. Vincular la función a un Trigger en la tabla movimientos
DROP TRIGGER IF EXISTS trigger_actualizar_kardex ON movimientos;
CREATE TRIGGER trigger_actualizar_kardex
AFTER INSERT ON movimientos
FOR EACH ROW
EXECUTE FUNCTION procesar_kardex_y_stock();
"""

# CAPA DE LÓGICA DE NEGOCIO
IGV_PORCENTAJE = 0.18

def calcular_igv(precio_base):
    """Calcula el monto del IGV (18%)"""
    if precio_base < 0:
        raise ValueError("El precio base no puede ser negativo.")
    return round(precio_base * IGV_PORCENTAJE, 2)


def calcular_precio_con_igv(precio_base):
    """Calcula el precio final sumando el IGV al precio base."""
    return round(precio_base + calcular_igv(precio_base), 2)


def calcular_margen_ganancia(precio_venta, costo_promedio):
    """Calcula el margen de ganancia en porcentaje."""
    if costo_promedio <= 0:
        raise ValueError("El costo promedio debe ser mayor a cero.")
    margen = ((precio_venta - costo_promedio) / costo_promedio) * 100
    return round(margen, 2)


def calcular_ganancia_item(precio_venta, costo_promedio, cantidad):
    """Calcula la ganancia bruta de vender una cantidad específica de un producto."""
    if cantidad <= 0:
        raise ValueError("La cantidad debe ser mayor a cero.")
    return round((precio_venta - costo_promedio) * cantidad, 2)


def validar_stock(stock_actual, cantidad_requerida):
    """Valida si hay suficiente stock para satisfacer una demanda."""
    return stock_actual >= cantidad_requerida


def detectar_productos_criticos(lista_productos):

    return [p for p in lista_productos if p.stock_actual <= p.stock_minimo]


def valorizar_inventario(lista_productos):

    total = sum(p.stock_actual * p.costo_promedio for p in lista_productos)
    return round(total, 2)

# Módulo 2: Clase Producto (POO)

class Producto:


    def __init__(self, id, nombre, sku, stock_actual, stock_minimo,
                 precio_venta, costo_promedio, categoria_id=None,
                 proveedor_id=None, unidad_id=None, fecha_vencimiento=None):
        self.id               = id
        self.nombre           = nombre
        self.sku              = sku
        self.stock_actual     = stock_actual
        self.stock_minimo     = stock_minimo
        self.precio_venta     = precio_venta
        self.costo_promedio   = costo_promedio
        self.categoria_id     = categoria_id
        self.proveedor_id     = proveedor_id
        self.unidad_id        = unidad_id
        self.fecha_vencimiento = fecha_vencimiento

    def esta_en_stock_critico(self):
        """Retorna True si el stock está en nivel de alerta (≤ stock mínimo)."""
        return self.stock_actual <= self.stock_minimo

    def valor_en_inventario(self):
        """Calcula el valor monetario de este producto en el inventario."""
        return round(self.stock_actual * self.costo_promedio, 2)

    def margen_ganancia(self):
        """Devuelve el margen de ganancia del producto usando la función pura."""
        return calcular_margen_ganancia(self.precio_venta, self.costo_promedio)

    def __str__(self):
        alerta = " ⚠️ CRÍTICO" if self.esta_en_stock_critico() else ""
        return (f"[{self.sku}] {self.nombre} | "
                f"Stock: {self.stock_actual} (mín: {self.stock_minimo}) | "
                f"Precio: S/. {self.precio_venta:.2f} | "

# Módulo 3: Clase Movimiento (POO)


class Movimiento:


    TIPOS_VALIDOS = ('ENTRADA', 'SALIDA')

    def __init__(self, producto, tipo_movimiento, cantidad, costo_operacion, motivo):

        self.producto         = producto
        self.tipo_movimiento  = tipo_movimiento.upper().strip()
        self.cantidad         = cantidad
        self.costo_operacion  = costo_operacion
        self.motivo           = motivo

    def validar(self):

        if self.tipo_movimiento not in self.TIPOS_VALIDOS:
            return False, (f"Tipo '{self.tipo_movimiento}' no es válido. "
                           f"Use 'ENTRADA' o 'SALIDA'.")

        if not isinstance(self.cantidad, int) or self.cantidad <= 0:
            return False, "La cantidad debe ser un número entero positivo."

        if self.costo_operacion < 0:
            return False, "El costo de operación no puede ser negativo."

        if not self.motivo or not self.motivo.strip():
            return False, "El motivo del movimiento no puede estar vacío."

        # Validación especial para salidas: verificar stock disponible
        if self.tipo_movimiento == 'SALIDA':
            if not validar_stock(self.producto.stock_actual, self.cantidad):
                return False, (
                    f"Stock insuficiente para '{self.producto.nombre}'. "
                    f"Disponible: {self.producto.stock_actual}, "
                    f"Solicitado: {self.cantidad}."
                )

        return True, "Movimiento válido."

    def ejecutar(self):

        es_valido, mensaje = self.validar()
        if not es_valido:
            print(f"  ❌ Movimiento rechazado: {mensaje}")
            return False, mensaje

        try:
            # Llamada a la función CRUD del Compañero 2 (sin SQL aquí)
            registrar_movimiento(
                self.producto.id,
                self.tipo_movimiento,
                self.cantidad,
                self.costo_operacion,
                self.motivo
            )

            # Actualizar el stock en memoria para reflejar el cambio
            if self.tipo_movimiento == 'ENTRADA':
                self.producto.stock_actual += self.cantidad
            else:
                self.producto.stock_actual -= self.cantidad

            return True, f"Movimiento '{self.tipo_movimiento}' registrado correctamente."

        except Exception as e:
            return False, f"Error al registrar en base de datos: {e}"

    def __str__(self):
        return (f"Movimiento {self.tipo_movimiento} | "
                f"Producto: {self.producto.nombre} | "
                f"Cantidad: {self.cantidad} | "
                f"Costo op.: S/. {self.costo_operacion:.2f} | "
                f"Motivo: {self.motivo}")


print(" Módulo 3: Clase Movimiento cargada correctamente.")

def crear_producto_en_bd(categoria_id, proveedor_id, unidad_id, nombre, sku,
                          stock_inicial, stock_minimo, precio_venta,
                          costo_inicial, fecha_vencimiento=None):

    if stock_inicial < 0 or stock_minimo < 0:
        raise ValueError("Los valores de stock no pueden ser negativos.")

    if precio_venta <= 0 or costo_inicial <= 0:
        raise ValueError("El precio de venta y el costo deben ser mayores a cero.")

    margen = calcular_margen_ganancia(precio_venta, costo_inicial)
    if margen < 0:
        print(f"  ⚠️  Advertencia: '{nombre}' se registra con margen negativo "
              f"({margen}%). Se vende por debajo del costo.")

    # Llamada al CRUD del Compañero 2 (sin SQL aquí)

        categoria_id, proveedor_id, unidad_id, nombre, sku,
        stock_inicial, stock_minimo, precio_venta,
        costo_inicial, fecha_vencimiento
    )

    # Devolver objeto Producto listo para usar en la lógica de negocio
    return Producto(
        id               = nuevo_id,
        nombre           = nombre,
        sku              = sku,
        stock_actual     = stock_inicial,
        stock_minimo     = stock_minimo,
        precio_venta     = precio_venta,

        categoria_id     = categoria_id,
        proveedor_id     = proveedor_id,
        unidad_id        = unidad_id,
        fecha_vencimiento= fecha_vencimiento
    )

def grafico_stock(productos):

    nombres = [p.nombre for p in productos]
    stock = [p.stock_actual for p in productos]

    plt.figure(figsize=(12,5))

    plt.bar(nombres, stock)

    plt.title(
        "Stock Actual por Producto",
        fontsize=16
    )

    plt.xlabel("Productos")
    plt.ylabel("Cantidad")

    plt.xticks(rotation=25)

    plt.grid(axis='y')

    plt.show()


def grafico_valorizacion(productos):

    nombres = [p.nombre for p in productos]

    valores = [
        p.valor_en_inventario()
        for p in productos
    ]

    plt.figure(figsize=(8,8))

    plt.pie(
        valores,
        labels=nombres,
        autopct='%1.1f%%'
    )

    plt.title(
        "Participación del Inventario",
        fontsize=16
    )

    plt.show()

def cargar_opciones_productos():
    try:
        conn = obtener_conexion()
        cur = conn.cursor()
        cur.execute("""
            SELECT id, nombre, sku, stock_actual,
                   stock_minimo, precio_venta, costo_promedio
            FROM productos;
        """)
        rows = cur.fetchall()
        cur.close()
        conn.close()
        return [(f"[{r[2]}] {r[1]}", r) for r in rows] if rows else [("Sin productos", None)]
    except:
        return [("Sin conexión", None)]

        with output:

            fila = dd_prod.value
            if fila is None:
                print("❌ Por favor, selecciona un producto.")
                return

            try:
                # Usamos el controlador para traer los datos (devuelve una lista de tuplas)
                registros = controller.obtener_historial_producto(fila[0])

                if not registros:
                    print(f"ℹ️ El producto '{fila[1]}' no tiene movimientos registrados.")
                    return

                # Encabezado de la tabla
                print(f"📋 Historial de movimientos: {fila[1]}")
                print("-" * 80)
                print(f"{'Fecha':<20} | {'Tipo':<12} | {'Cant':<6} | {'Costo':<10} | {'Motivo'}")
                print("-" * 80)

                # Iteramos sobre las tuplas usando índices
                # Posiciones: [0]fecha, [1]tipo_movimiento, [2]cantidad, [3]costo_operacion, [4]motivo
                for r in registros:
                    fecha  = str(r[0])[:19]
                    tipo   = str(r[1])
                    cant   = str(r[2])
                    costo  = f"S/.{float(r[3]):.2f}"
                    motivo = str(r[4])

                    print(f"{fecha:<20} | {tipo:<12} | {cant:<6} | {costo:<10} | {motivo}")

                print("-" * 80)
                print(f"✅ Total registros encontrados: {len(registros)}")

            except Exception as e:
                print(f"❌ Error al consultar movimientos: {e}")

    btn_ver.on_click(al_consultar)
